Developer API

Programmatic access to design tokens and template generation

Token Extraction

// Extract design tokens from existing templates
curl -X POST https://api.stylestack.io/v1/extract \
  -H "Content-Type: multipart/form-data" \
  -F "file=@presentation.pptx" \
  -F "format=json" \
  -F "extract_assets=true"

// Response
{
  "stylestack": {
    "version": "1.0.0",
    "tokens": {
      "colors": {
        "primary": "#0066CC",
        "accent": "#FF6600"
      },
      "typography": {
        "font_family": "Inter, sans-serif",
        "sizes": {
          "heading": "24pt",
          "body": "12pt"
        }
      },
      "brand_assets": {
        "logos": {...},
        "icons": {...}
      }
    }
  }
}

Template Generation

// Generate templates with design tokens
const response = await fetch('/api/v1/generate', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({
    template: 'presentation-corporate',
    organization: 'acme-corp',
    channel: 'quarterly-reports',
    tokens: {
      brand_primary: '#0066CC',
      brand_font: 'Inter',
      logo_url: 'https://cdn.acme.com/logo.svg'
    },
    formats: ['pptx', 'odp', 'google-slides']
  })
});

// Returns multi-format template bundle
const { templates } = await response.json();
